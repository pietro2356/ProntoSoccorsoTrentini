import { ChangeDetectionStrategy, Component, input } from '@angular/core';
import { IonicModule } from '@ionic/angular';

@Component({
  selector: 'pst-error-card',
  templateUrl: './error-card.component.html',
  styleUrls: ['./error-card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
  imports: [IonicModule],
})
export class ErrorCardComponent {
  title = input<string>();
  message = input<string>();
  error = input<string>();
  isError = input.required<boolean>();
}

<ion-card>
  <ion-card-header>
    <ion-card-title>
      @if (title()){
        {{title()}}
      }
      @else {
      🚨Dati dei Pronto Soccorso non disponibili
      }
    </ion-card-title>
  </ion-card-header>
  <ion-card-content>
    @if (isError()){
      <p>Il servizio di Pronto Soccorso non è disponibile al momento. Riprovare più tardi.</p>
      @if(error()){
        <br>
        <p>Errore: {{error()}}</p>
      }
    } @else if (!isError()) {
      @if (message()){
        <p>{{message()}}</p>
      } @else {
        <p>Non ci sono dati disponibili</p>
      }
    }
  </ion-card-content>
</ion-card>
